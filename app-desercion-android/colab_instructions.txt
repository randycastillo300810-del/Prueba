# CELDA 3: Subir archivos (Ejecuta y sube tu proyecto cuando te lo pida)
from google.colab import files
import zipfile
import os

# Sube tu proyecto como ZIP
uploaded = files.upload()

# Descomprimir
for filename in uploaded.keys():
    with zipfile.ZipFile(filename, 'r') as zip_ref:
        zip_ref.extractall('.')
